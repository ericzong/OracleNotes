创建


create sequence <SequenceName>
  [increment by <StepSize>]		-- 默认 1
  [start with <StartValue>]		-- 默认 1
  [minvalue <MinValue>|nominvalue]	-- 默认 1
  [maxvalue <MaxValue>|nomaxvalue]	-- 默认 10^27
  [cycle|nocycle]
  [cache <CacheSize>|nocache]		-- 默认 20
  [order|noorder];

修改


alter sequence <SequenceName>
  [increment by <StepSize>]
  [minvalue <MinValue>|nominvalue]
  [maxvalue <MaxValue>|nomaxvalue]
  [cycle|nocycle]
  [cache <CacheSize>|nocache]
  [order|noorder];

删除


drop sequence <SequenceName>;
