创建


create role <Rolename>
  [not <identified|identified> by <Password>];

修改


alter role <Rolename>
  [not <identified|identified> by <Password>];

删除


drop role <Password>;

赋予


grant <Rolename> to <Username> [with admin option];
grant <Rolename> to <OtherRoleName> [with admin option];
