创建


--创建表空间
create tablespace <tablespace_name>
  datafile <datafile_path>
  size <file_size>
  autoextend on
  next <next_size> maxsize [<max_size> | unlimited]
  extent management local;

--创建临时表空间
create temporary tablespace <tablespace_name>
  tempfile <tempfile_path>
  size <file_size>
  autoextend on
  next <next_size> maxsize <max_size>
  extent management local;
  
--创建 UNDO 表空间
create undo tablespace <tablespace_name>
  datafile <datafile_path>
  size <file_size>
  autoextend on
  next <next_size> maxsize [<max_size> | unlimited]
  extent management local;

删除


drop tablespace <tablespace_name> including contents and datafiles;

修改


--修改默认表空间
alter [user <user_name>|database] default tablespace <tablespace_name>;
alter [user <user_name>|database] default temporary tablespace <tablespace_name>;
alter system set undo_tablespace='';

--修改表空间名称
alter tablespace <tablespace_name> rename to <new_tablespace_name>;

--修改表空间数据文件配置
alter database datafile <datafile_path>
  autoextend on next <next_size> maxsize [<max_size> | unlimited];
alter database datafile <file_path> resize <size>;
alter database tempfile <tempfile_path> resize <size>;
  
--修改表空间对应的数据文件
alter tablespace <tablespace_name> add|drop datafile <file_path> size <size>
  autoextend on next <extend_size>;

alter tablespace <tablespace_name> [online | offline];

alter tablespace <tablespace_name> [read only | write];

查询


--查询默认表空间
select username, default_tablespace from dba_users;
show parameter undo;

--查询个表空间文件信息
select file_name, tablespace_name, bytes from dba_data_files;

--查询表空间包含的对象
select t.owner, t.segment_name, sum(bytes) / 1024 / 1024
  from dba_segments t
 where t.tablespace_name = <tablespace_name>
 group by t.owner, t.segment_name
 order by sum(bytes) desc;

相关

移动表


alter table <table_name> move tablespace <tablespace_name>;

移动索引


--移动普通索引
alter index <index_name> rebuild tablespace <tablespace_name>;

技巧

批量生成转移语句


--批量生成表转移语句，将某些表转移到指定的表空间中
select 'alter table ' || t.table_name || ' move tablespace <tablespace_name>;' stm
  from user_tables t
 where t.tablespace_name = <from_tablespace_name>
 order by t.table_name;

select 'alter table ' || t.segment_name || ' move tablespace <tablespace_name>;' stm, sum(t.BYTES/1024/1024) "size(M)"
  from dba_segments t
 where t.tablespace_name = <from_tablespace_name>
   and t.owner = <user_name>
   and t.segment_type = 'TABLE'
 group by t.segment_name
 order by segment_name; 
 
--批量生成索引转移语句，将某些索引转移到指定的表空间中
select 'alter index ' || t.index_name || ' rebuild tablespace <tablespace_name>;' stm
  from user_indexes t
 where t.tablespace_name = <user_name>
 order by t.index_name;

select 'alter index ' || t.segment_name || ' rebuild tablespace <tablespace_name>;' stm, sum(t.BYTES/1024/1024) "size(M)"
  from dba_segments t
 where t.tablespace_name = <from_tablespace_name>
   and t.owner = <user_name>
   and t.segment_type = 'INDEX'
 group by t.segment_name
 order by segment_name;
推荐从 user_tables、user_indexes 生成，因为并非所有表、索引都会在 dba_segments 中，dba_segments 优点在于可以统计占用空间的大小（可用以评估移动的耗时）。
