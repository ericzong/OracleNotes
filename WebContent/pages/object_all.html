清除


purge table <table_name>;
purge index <index_name>;
--一般用户清除自己的回收站
purge recyclebin;
--dba清除任何表空间中的回收站
purge dba_recyclebin;
--清除指定表空间回收站
purge tablespace <ts_name>;
--清除指定表空间中指定用户的回收站
purge tablespace <ts_name> user <user_name>;
