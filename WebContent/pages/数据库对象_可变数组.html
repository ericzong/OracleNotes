create type worker_type as object
(
  id   number,
  name varchar2(32)
);

create type worker_list as varray(20) of worker_type;

create table factory(id number, name varchar2(32), workers worker_list);

insert into factory values(1, '工厂1 ', worker_list(worker_type(1, '张三'), worker_type(2, '李四'), worker_type(3, '王五')));

用法跟嵌套表相似，与嵌套表不同在于：

  1. 可变数组的列不会自动加unique index;

  2. 创建表时不必指定可变数组的存储位置（也不会自动创建存储表，尚不明确可变数组的存储位置）
