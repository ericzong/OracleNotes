运行

dbms_scheduler.run_job([name=>]<JOB_NAME>);

停止

dbms_scheduler.stop_job('<JOB_NAME, SYS.DEFAULT_JOB_CLASS>');

禁/启用

dbms_scheduler.disable([name=>]<JOB_NAME>);
dbms_scheduler.enable([name=>]<JOB_NAME>);

查询

select state, t.* from user_scheduler_jobs t;
select * from user_scheduler_running_jobs;
